🔐 Firebase Security Rules
قواعد الحماية والأمان
Firestore Rules — انسخها والصقها في Firebase Console
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ✅ دالة للتحقق من تسجيل الدخول
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // ✅ دالة للتحقق من دور المدير العام
    function isSuperAdmin() {
      return isAuthenticated() && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // ✅ دالة للتحقق من مدير الفرع
    function isBranchManager(branchId) {
      return isAuthenticated() && (
        isSuperAdmin() ||
        (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.branchId == branchId &&
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['admin', 'manager'])
      );
    }
    
    // ✅ التحقق من أن المستخدم ينتمي للفرع
    function belongsToBranch(branchId) {
      return isAuthenticated() && (
        isSuperAdmin() ||
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.branchId == branchId
      );
    }

    // 🔒 إعدادات النظام — المدير العام فقط
    match /system/{docId} {
      allow read: if isAuthenticated();
      allow write: if isSuperAdmin();
    }

    // 🔒 بيانات المستخدمين
    match /users/{userId} {
      allow read: if isAuthenticated() && (request.auth.uid == userId || isSuperAdmin());
      allow create: if isSuperAdmin();
      allow update: if isSuperAdmin() || request.auth.uid == userId;
      allow delete: if isSuperAdmin();
    }

    // 🔒 بيانات الفروع
    match /branches/{branchId} {
      allow read: if isAuthenticated();
      allow write: if isSuperAdmin();
    }

    // 🔒 صلاحيات الأدوار — المدير العام فقط
    match /roles/{roleId} {
      allow read: if isAuthenticated();
      allow write: if isSuperAdmin();
    }
    
    // 🔒 سجل الأحداث
    match /activityLog/{logId} {
      allow read: if isSuperAdmin();
      allow create: if isAuthenticated();
      allow update, delete: if isSuperAdmin();
    }

    // 🔒 بيانات كل فرع (مبيعات، مخزون، عملاء...)
    match /branches/{branchId}/{collection}/{docId} {
      allow read: if belongsToBranch(branchId);
      allow write: if belongsToBranch(branchId);
    }
    
    // 🔒 التقارير الموحدة
    match /reports/{reportId} {
      allow read, write: if isSuperAdmin();
    }
  }
}

1
افتح Firebase Console
console.firebase.google.com → مشروعك
2
Firestore Database → Rules
من القائمة الجانبية اختر Firestore ثم تبويب Rules
3
الصق القواعد أعلاه
استبدل المحتوى الافتراضي بالقواعد المنسوخة
4
Publish
